def run_simulation_numpy(num_people):

    #  NumPy 
    ids = np.arange(1, num_people + 1)
    killed_turn = np.full(num_people, -1)  # 
    turn_num = num_people

    for turn in range(turn_num):
        alive_ids = ids[killed_turn == -1]  # ID
        if len(alive_ids) == 0:  # 
            break

        random_odd_idx = random.randrange(1, len(alive_ids) + 1, 2)
        kill_id = alive_ids[random_odd_idx - 1] - 1  # ID

        killed_turn[kill_id] = turn + 1  # 

    return killed_turn
