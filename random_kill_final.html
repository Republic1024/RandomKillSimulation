from tqdm import tqdm_notebook


def random_kill(n):
    # 0n-1
    return random.choice([i for i in range(n) if i % 2 == 0])


def random_kill_list(alive_ids):
    n = len(alive_ids)
    return alive_ids[random_kill(n)]


def run_simulation_numpy(num_people):

    #  NumPy 
    ids = np.arange(1, num_people + 1)
    status = np.ones(num_people, dtype=int)  # 1
    killed_turn = np.full(num_people, -1)  # 

    for turn in range(num_people):
        alive_ids = ids[status == 1]  # ID
        if len(alive_ids) == 0:  # 
            break

        kill_id = random_kill_list(alive_ids) - 1  # ID

        status[kill_id] = 0  # 
        killed_turn[kill_id] = turn + 1  # 

    #  DataFrame
    df = pd.DataFrame({
        "id": ids,
        "status": status,
        "killed_turn": killed_turn
    })

    return df
